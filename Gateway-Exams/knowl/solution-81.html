\begin{align*} \int \left(\frac{1}{2}x^2 - 2\sqrt{x} +3\right) \, dx \amp = \int \left( \frac{1}{2} x^2 - 2x^{1}{2} + 3\right) \, dx\\ \amp = \frac{1}{2} \cdot \frac{x^3}{3} - 2\frac{x^{3/2}}{3/2} + 3x + C\\ \amp = \frac{x^3}{6} - \frac{4}{3} x^{3/2} + 3x + C \end{align*}
